  <div id="container" tabindex="0" class="blockquote text-nowrap">
    
  </div>
  <script>
        var container = document.getElementById('container');
      window.onload = function() {
        container.textContent = "";
        add_new_chars();
      }
      window.addEventListener("keyup", function (e) {
        console.log(e.key);
        if (e.key == container.textContent[0]) {
          container.textContent = "";
        }
        add_new_chars();
      })
        function add_new_chars() {
        for(var i = 0; i < Math.random()*3; i++) {
        var CHAR = (Math.floor(Math.random()*26));
        container.textContent += String.fromCharCode(CHAR + 97);
        }
    };
  </script>
